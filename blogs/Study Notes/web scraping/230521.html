    def download(url, num_retries=2):
    print('Downloading:', url)
    # 设置用户代理
    user_agent = '马赛克'
    headers = {'User-agent': user_agent}
    request = urllib.request.Request(url, headers=headers)  # 访问url时附带请求头UA信息
    try:
        response = urllib.request.urlopen(request)
        html = response.read().decode('utf-8')
        print('Download successfully')
    except urllib.error.URLError as e:
        print('Download error:', e.reason)
        html = None
        # 下载失败重试
        if num_retries > 0:
            if hasattr(e, 'errno') and 500 <= e.errno < 600:
                return download(url, num_retries - 1)
    return html # 类型:str

    # 分析网站信息

    • robots.txt: 大多数网站定义了该文件，让爬虫了解网站存在的限制。

    • Sitemap: 网站地图，帮助爬虫定位网站最新的内容而无须爬取每一个网页。可以从robots.txt文件中获取Sitemap。

    " 根域名 + /robots.txt " 获取网站的robots.txt文件

    Crawl-delay字段定义了抓取延迟(s)，爬取过快可能会导致IP被封禁等后果

    对爬虫采取限速操作，对同一网站的两次爬取间隔必须大于 Crawl-delay规定的时间。

import time
import urllib.parse
import datetime

    class Throttle:
   """
       限速类
   """

   # 构造函数
   def __init__(self, delay):
       self.delay = delay
       self.domains = {}  # 公有属性，字典类型，域名-上次访问时间

   def wait(self, url):
       domain = urllib.parse.urlparse(url).netloc  # 获取域名服务器位置
       last_accessed = self.domains.get(domain)    # 获取上次访问该域名的时间
       if self.delay > 0 and last_accessed is not None:
           sleep_secs = self.delay - (datetime.datetime.now() - last_accessed).seconds  # 计算休眠时间
           if sleep_secs > 0:
               time.sleep(sleep_secs)
       self.domains[domain] = datetime.datetime.now()

    def crawler_bs(datalist, html):
    soup = BeautifulSoup(html, 'html.parser')

    for item in soup.find_all('div', attrs={'class': 'info'}):
        data = []

        title = item.find('span', attrs={'class': 'title'})
        if title is not None:
            data.append(title.text)
        else:
            data.append("")

        rating_num = item.find('span', attrs={'class': 'rating_num'})
        if rating_num is not None:
            data.append(rating_num.text)
        else:
            data.append("")

        quote = item.find('span', attrs={'class': 'inq'})
        if quote is not None:
            data.append(quote.text)
        else:
            data.append("")

        info = item.find('p', attrs={'class': ""})
        if info is not None:
            string = re.sub(r'\n', "", info.text)   # 删除空行
            string = re.sub(r'  +', "", string)     # 将多个空格替换成一个空格
            string = string.replace('/', ' ')       # 将斜线替换成一个空格
            string = string.replace(u'\xa0', u' ')  # 将&nbsp替换成一个空格
            data.append(string.strip())
        else:
            data.append("")
        datalist.append(data)

    def crawler_lxml(datalist, html):
    tree = etree.HTML(html)

    item_list = tree.xpath('//div[@class="item"]')
    for item in item_list:
        data = []
        title = item.xpath('.//span[@class="title"]/text()')
        if title is not None:
            data.append(title[0])
        else:
            data.append("")
        rating_num = item.xpath('.//span[@class="rating_num"]/text()')
        if rating_num is not None:
            data.append(rating_num[0])
        else:
            data.append("")
        quote = item.xpath('.//span[@class="inq"]/text()')
        if quote is not None:
            data.append(quote[0])
        else:
            data.append("")

        info = item.xpath('.//p[@class=""]/text()')
        if info is not None:
            string = re.sub(r'\n', "", info[0])  # 删除空行
            string = re.sub(r'  +', "", string)  # 将多个空格替换成一个空格
            string = string.replace('/', ' ')  # 将斜线替换成一个空格
            string = string.replace(u'\xa0', u' ')  # 将&nbsp替换成一个空格
            data.append(string.strip())
        else:
            data.append("")
        print(data)
        datalist.append(data)

    def saveData(datalist, savepath):
    print("saving.......")
    book = xlwt.Workbook(encoding="utf-8", style_compression=0)  # 创建workbook对象
    sheet = book.add_sheet('豆瓣电影top250', cell_overwrite_ok=True)  # 创建工作表
    col = ("影片名", "评分", "语录", "相关信息")
    for i in range(0, 4):
        sheet.write(0, i, col[i])  # 0:行，i:列，col[i]:列名
    for i in range(0, 250):
        data = datalist[i]
        for j in range(0, 4):
            sheet.write(i + 1, j, data[j])  # data[j]数据
    book.save(savepath)  # 保存
    print("save successfully")


def main():
    baseurl = "https://movie.douban.com/top250?start="  # 要爬取的网页链接
    datalist = []
    # 每页只有25个电影，共10页
    throttle = Throttle(5)  # 两次爬取之间间隔5s
    for i in range(0, 10):
        url = baseurl + str(i * 25)
        throttle.wait(url)
        html = download(url)
        crawler(datalist, html)
    saveData(datalist, "豆瓣电影top250.xls")
